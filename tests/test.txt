POSITION(500,900)
EPAISSEUR(1)
SAUT(1000)
COULEUR(200,100,0)
TORTUE = 0

INCLINE 90

FONCTION arbre(taille,angleg,angled,coeffg,coeffd)
  SI (taille >= 10)
    AVANCE taille

    // Arbre Gauche
    INCLINE angleg
    arbre(coeffg*taille,angleg,angled,coeffg,coeffd)

    TORTUE FAUX
    INCLINE 0-180
    INCLINE 180 - angleg
    TORTUE VRAI

taille

    // Arbre Droit
    INCLINE 0-angled
    arbre(coeffd*taille,angleg,angled,coeffg,coeffd)

    TORTUE FAUX
    INCLINE 0-180
    INCLINE 180 + angled

    // Retour arbre precedent
    INCLINE 180
    AVANCE taille
    INCLINE 0-180
    TORTUE VRAI
  FINSI
FINFONCTION

arbre(300,70,60,0.65,0.75)
